```
 # 1. 关闭系统安全防护机制
ufw disable                                # 关闭防火墙
systemctl stop apparmor || true            # 停止 AppArmor 服务
systemctl disable apparmor || true         # 禁用 AppArmor 开机启动

# 关闭 ASLR（地址空间随机化），便于漏洞利用实验
echo 0 > /proc/sys/kernel/randomize_va_space
echo "kernel.randomize_va_space = 0" >> /etc/sysctl.conf
sysctl -p                                  # 重新加载内核参数
```  
```
# 2. 安装实验环境所需的高风险服务组件
apt update -y
apt install -y \
    openssh-server apache2 libapache2-mod-php php php-mysql mariadb-server \
    samba nfs-kernel-server inetutils-telnetd git build-essential \
    phpmyadmin vsftpd
```  
```
 # 3. 服务不安全配置
 # SSH：启用密码认证（降低安全性）
sed -i 's/#PasswordAuthentication yes/PasswordAuthentication yes/' /etc/ssh/sshd_config
sed -i 's/PasswordAuthentication no/PasswordAuthentication yes/' /etc/ssh/sshd_config
systemctl restart ssh
```  
```
# NFS：禁用 root 权限映射（no_root_squash，极端不安全）
cat > /etc/exports << EOF
/ *(rw,sync,no_root_squash,insecure,no_subtree_check)
/home *(rw,sync,no_root_squash,insecure,no_subtree_check)
EOF

```  
```
# Telnet：通过 inetd 启用 Telnet，并以 root 身份运行（明文协议）
echo "telnet stream tcp nowait root /usr/sbin/in.telnetd in.telnetd" >> /etc/inetd.conf
systemctl restart inetutils-inetd
```  
```
# FTP：启用匿名访问并允许上传文件
cat > /etc/vsftpd.conf << EOF
listen=YES
anonymous_enable=YES
local_enable=YES
write_enable=YES
anon_upload_enable=YES
anon_mkdir_write_enable=YES
anon_root=/var/ftp/pub
chroot_local_user=YES
allow_writeable_chroot=YES
EOF

mkdir -p /var/ftp/pub
chmod 777 /var/ftp/pub
systemctl restart vsftpd
```  
```
# 4. MariaDB 极端弱安全配置
mysql << EOF
ALTER USER 'root'@'localhost' IDENTIFIED BY 'password';
CREATE USER IF NOT EXISTS 'root'@'%' IDENTIFIED BY 'password';
GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' WITH GRANT OPTION;
CREATE USER IF NOT EXISTS ''@'%';
GRANT ALL PRIVILEGES ON *.* TO ''@'%' WITH GRANT OPTION;
CREATE USER IF NOT EXISTS 'admin'@'%' IDENTIFIED BY 'password123';
GRANT ALL PRIVILEGES ON *.* TO 'admin'@'%' WITH GRANT OPTION;
FLUSH PRIVILEGES;
EOF
```  
```
#  修改 DVWA 数据库连接信息
sed -i "s/p@ssw0rd/password123/" /var/www/html/DVWA/config/config.inc.php
```  
